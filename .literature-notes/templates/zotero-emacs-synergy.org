#+begin_src elisp
(use-package citar
  :custom
  (org-cite-global-bibliography '("~/knowledge/episteme/citations.bib"))
  (citar-bibliography org-cite-global-bibliography)
  (citar-library-paths '("~/Zotero/storage/"))      ; the PDFs
  (citar-notes-paths   '("~/knowledge/episteme/.literature-notes/"))
  (org-cite-insert-processor 'citar)
  (org-cite-follow-processor 'citar)
  (org-cite-activate-processor 'citar)
  :bind (:map org-mode-map ("C-c b" . #'org-cite-insert)))

(use-package citar-org-roam
  :after (citar org-roam)
  :config (citar-org-roam-mode))
#+end_src
- C-c b           insert citation
- citar-open      open a reference's PDF / URL / note
- citar-open-notes  create/open the org-roam lit-note
